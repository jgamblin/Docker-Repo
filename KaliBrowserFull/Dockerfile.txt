FROM kalilinux/kali-rolling:2024.1
LABEL maintainer="Light Ning <lightning1141@gmail.com>"

# Set correct environment variables
ENV HOME /root/
ENV DEBIAN_FRONTEND noninteractive
ENV LC_ALL C.UTF-8
ENV LANG en_US.UTF-8
ENV LANGUAGE en_US.UTF-8

RUN apt-get update && \
    apt-get install -y --no-install-recommends \
    	kali-linux-full \
        ca-certificates \
        git \
        xvfb \
        x11vnc \
        xdotool \
        wget \
        python \
        python-numpy \
        unzip \
        firefox \
        openbox \
        menu \
        supervisor \
        x11-apps \
        net-tools \
        lightdm \
        curl \
    && rm -rf /var/lib/apt/lists/*

ENV DISPLAY :0.0

WORKDIR /root
RUN git clone https://github.com/kanaka/noVNC.git

WORKDIR /root/noVNC/utils
RUN git clone https://github.com/kanaka/websockify websockify

WORKDIR /root

COPY supervisord.conf /etc/supervisor/conf.d/supervisord.conf

# Create non-root user
RUN useradd -m -u 1000 kaliuser && \
    chown -R kaliuser:kaliuser /root
USER kaliuser

# Add healthcheck
HEALTHCHECK --interval=30s --timeout=3s --start-period=10s --retries=3 \
    CMD curl -f http://localhost:6080/ || exit 1

# Expose Port
EXPOSE 6080

CMD ["/usr/bin/supervisord"]
